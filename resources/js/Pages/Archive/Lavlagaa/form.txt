const handleDownload = async () => {
  const response = await axios.post(
    "/api/generate-doc",
    { content: textValue },
    { responseType: "blob" }
  );

  const url = window.URL.createObjectURL(new Blob([response.data]));
  const link = document.createElement("a");
  link.href = url;
  link.download = "document.docx";
  link.click();
};
